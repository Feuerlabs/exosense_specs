HOWTO
=====

# 1- create the an account, the admin user is create automatically by script

create-account.sh <name> <password>

# 2 - create a device type

create-device-type.sh <device-type-name> <protocol>

protocol: ga_ck3 | exodm_bert ...

# 3a - upload yang spec 

create-yang-module <yang-spec-file-name>

# 3 - create config set

create-config-set.sh <config-set-name> <yang-spec-name> <notification-url>

# 4 - provision device

provision-device.sh <device-id> <device-type-name> <config-set-name> 
		    <server-key> <device-key>



# example bootstrap of dm server

create-account.sh abc abc123

create-device-type.sh v1 exodm_bert

create-yang-module.sh spec.yang

create-config-set.sh conf1 spec.yang http://localhost:8380

provision-device.sh 1234 v1 conf1 1 2
provision-device.sh 1235 v1 conf1 1 2
provision-device.sh 1236 v1 conf1 1 2
